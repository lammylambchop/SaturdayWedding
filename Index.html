  document.getElementById('task-time').style.borderColor  = time  ? '' : 'var(--rose)';

  document.getElementById('task-title').style.borderColor = title ? '' : 'var(--rose)';

  if (!time || !title) return;

  if (editId) {

    var idx = tasks.findIndex(function(t){return String(t.id)===editId;});

    if (idx !== -1) {

      tasks[idx] = Object.assign({}, tasks[idx], {time:time, title:title, note:note});

      if (pendingPhotos.length) { taskPhotos[tasks[idx].id] = (taskPhotos[tasks[idx].id]||[]).concat(pendingPhotos); savePhotos(); }

    }

  } else {

    var id = Date.now();

    tasks.push({id:id, time:time, title:title, note:note});

    if (pendingPhotos.length) { taskPhotos[id] = pendingPhotos.slice(); savePhotos(); }

  }

  tasks.sort(function(a,b){return timeToMins(a.time)-timeToMins(b.time);});

  save(); resetForm(); render();

}


function editTask(id) {

  var t = tasks.find(function(t){return t.id===id;});

  if (!t) return;

  document.getElementById('editing-id').value = id;

  document.getElementById('task-time').value  = t.time;

  document.getElementById('task-title').value = t.title;

  document.getElementById('task-note').value  = t.note || '';

  pendingPhotos = taskPhotos[id] ? taskPhotos[id].slice() : [];

  renderFormPreviews();

  document.getElementById('form-heading').textContent = 'Edit Task';

  document.getElementById('submit-btn').textContent   = 'Save Changes';

  document.getElementById('cancel-btn').style.display = 'inline-block';

  document.getElementById('task-time').scrollIntoView({behavior:'smooth', block:'center'});

}


function removeTask(id) {

  if (!confirm('Remove this task?')) return;

  tasks = tasks.filter(function(t){return t.id!==id;});

  delete taskPhotos[id];

  save(); savePhotos(); render();

}

}


function clearAllTasks() {

  if (!confirm('Clear all tasks and start fresh?')) return;

  tasks = []; taskPhotos = {}; save(); savePhotos(); render();

}


function toggleReminders(on) {

  remindersOn = on;

  if (on) { tick(); tickTimer = setInterval(tick, 30000); }

  else { clearInterval(tickTimer); document.getElementById('alarm-strip').style.display='none'; }

}


function tick() {

  render();

  if (!remindersOn) return;

  tasks.forEach(function(task) {

    var diff = minsUntil(task.time);

    [0, 15].forEach(function(ahead) {

      var key = task.id + '-' + ahead;

      var inWindow = ahead===0 ? (diff>=-1 && diff<=1) : (diff>=ahead-1 && diff<=ahead+1);

      if (inWindow && !dismissedAlarms[key]) {

        dismissedAlarms[key] = true;

        showAlarm(task, ahead);

      }

    });

  });

}


function showAlarm(task, ahead) {

  document.getElementById('alarm-title').textContent = ahead===0 ? 'Now: '+task.title : 'In 15 mins: '+task.title;

  document.getElementById('alarm-sub').textContent   = ahead===0 ? "It's time! ("+formatTime(task.time)+')' : 'Coming up at '+formatTime(task.time);

  document.getElementById('alarm-strip').style.display = 'block';

  window.scrollTo({top:0, behavior:'smooth'});

  if (navigator.vibrate) navigator.vibrate([300,100,300]);

}


function dismissAlarm() { document.getElementById('alarm-strip').style.display='none'; }


function render() {

  var current = tasks.find(function(t){return taskStatus(t)==='current';});

  var banner  = document.getElementById('current-task-banner');

  if (current) {

    banner.style.display = 'block';

    document.getElementById('banner-task-name').textContent = current.title;

    document.getElementById('banner-task-time').textContent = 'Started at ' + formatTime(current.time);

  } else { banner.style.display='none'; }


  var next = tasks.find(function(t){return minsUntil(t.time)>1;});

  var chip = document.getElementById('next-chip');

  if (next) {

    var diff = minsUntil(next.time);

    var label = diff<60 ? diff+'min'+(diff!==1?'s':'') : Math.floor(diff/60)+'h '+(diff%60>0?diff%60+'m':'');

    chip.style.display='block';

    document.getElementById('next-label').textContent    = next.title;

    document.getElementById('next-countdown').textContent = label;

  } else { chip.style.display='none'; }


  var tl = document.getElementById('timeline');

  if (!tasks.length) { tl.innerHTML='<div class="empty-state"><div class="icon">🌸</div><p>Add your first task above</p></div>'; return; }


  tl.innerHTML = tasks.map(function(task) {

    var status = taskStatus(task);

    var diff   = minsUntil(task.time);

    var badge  = status==='current' ? '<span class="badge badge-now">Now</span>'

               : status==='done'    ? '<span class="badge badge-done">Done</span>'

               : diff<=30           ? '<span class="badge badge-soon">Soon</span>' : '';

    var countdown = '';

    if (status==='upcoming') {

      countdown = diff<60 ? '<div class="countdown-chip">in '+diff+' min'+(diff!==1?'s':'')+'</div>'

                          : '<div class="countdown-chip">in '+Math.floor(diff/60)+'h '+(diff%60>0?diff%60+'m':'')+'</div>';

    }

    var photos = taskPhotos[task.id] || [];

    var photoHtml = '<div class="task-photos">'

      + photos.map(function(p){ return '<img src="'+p.dataUrl+'" onclick="openLightbox(\''+p.dataUrl+'\')">'; }).join('')

      + '<button class="task-add-photo-btn" onclick="pickTaskPhoto('+task.id+')" title="Add photo">+</button>'

      + '</div>';


    return '<div class="task-item '+status+'">'

      +'<div class="time-col">'+formatTime(task.time)+'</div>'

      +'<div class="line-col"><div class="dot"></div><div class="connector"></div></div>'

      +'<div class="content-col">'

        +'<div class="task-title-row"><div class="task-title">'+task.title+'</div>'+badge+'</div>'

        +(task.note?'<div class="task-note">'+task.note+'</div>':'')

        +countdown

        +photoHtml

        +'<div class="task-action-row">'

          +'<button class="task-btn" onclick="editTask('+task.id+')">Edit</button>'

          +'<button class="task-btn del" onclick="removeTask('+task.id+')">Remove</button>'

        +'</div>'

      +'</div>'

    +'</div>';

  }).join('');

}

function exportData() {

  // Export JSON for backup

  var data = { tasks: tasks, taskPhotos: taskPhotos };

  var json = JSON.stringify(data);

  var blob = new Blob([json], {type: 'application/json'});

  var url  = URL.createObjectURL(blob);

  var a    = document.createElement('a');

  a.href     = url;

  a.download = 'wedding-schedule.json';

  a.click();

  URL.revokeObjectURL(url);

}


function exportHTML() {

  // Bake current tasks into a standalone HTML file

  var data = JSON.stringify({ tasks: tasks, taskPhotos: taskPhotos });

  var injected = '<script id="baked-data" type="application/json">' + data + '<\/script>';

  // Get current page HTML and inject the baked data just before </body>

  var html = document.documentElement.outerHTML;

  html = html.replace('</body>', injected + '</body>');

  var blob = new Blob([html], {type: 'text/html'});

  var url  = URL.createObjectURL(blob);

  var a    = document.createElement('a');

  a.href     = url;

  a.download = 'wedding-day-reminder.html';

  a.click();

  URL.revokeObjectURL(url);

}

function importFromText() {

  var text = document.getElementById('import-text').value.trim();

  if (!text) { alert('Please paste your schedule data first.'); return; }

  try {

    var data = JSON.parse(text);

    if (data.tasks && data.tasks.length) {

      tasks      = data.tasks;

      taskPhotos = data.taskPhotos || {};

      save();

      savePhotos();

      render();

      hideImportBox();

      alert('Done! ' + tasks.length + ' tasks loaded.');

    } else {

      alert('No tasks found in the pasted data. Make sure you copied the full contents of your wedding-schedule.json file.');

    }

  } catch(err) {

    alert('Could not read the pasted text. Make sure you copied the full contents of the wedding-schedule.json file without any changes.');

  }

}

        savePhotos();

      }

      load(); render(); maybeShowInstallTip();

    })

    .catch(function() {

      // JSON not found or error — fall back to localStorage

      load(); render(); maybeShowInstallTip();

    });

})();
